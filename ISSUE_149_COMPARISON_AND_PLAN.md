class BotOrchestrator:
    def __init__(self):
        self.grid_engine = GridEngine()
        self.dca_engine = DCAEngine()
        self.risk_manager = RiskManager()
        self.active_bots = {}

    async def execute_signal(self, aggregated_signal):
        # Risk check
        if not self.risk_manager.can_open_position(signal):
            return

        # Strategy selection
        if signal.strategy == "GRID":
            bot = await self.grid_engine.create_bot(
                lower_price=signal.lower_bound,
                upper_price=signal.upper_bound,
                grid_levels=signal.grid_levels,
                amount_per_grid=self.calculate_position_size(signal)
            )

        elif signal.strategy == "DCA":
            bot = await self.dca_engine.create_bot(
                entry_price=signal.entry,
                dca_steps=signal.dca_steps,
                amount_per_step=self.calculate_position_size(signal),
                take_profit=signal.tp
            )

        elif signal.strategy == "HYBRID":
            bot = await self.create_hybrid_bot(signal)

        # Track bot
        self.active_bots[bot.id] = bot

        # Monitor and adapt
        await self.monitor_bot(bot)
